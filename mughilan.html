Sentiment & Emotion Detection in Social Media Conversations

pip install transformers torch snscrape pandas matplotlib

Python Source Code

python
import pandas as pd
import snscrape.modules.twitter as sntwitter
from transformers import pipeline
import matplotlib.pyplot as plt

# -------- 1. Scrape Twitter data --------
def scrape_tweets(query, max_tweets=100):
    tweets = []
    for i, tweet in enumerate(sntwitter.TwitterSearchScraper(query).get_items()):
        if i > max_tweets:
            break
        tweets.append(tweet.content)
    return tweets

# -------- 2. Load emotion classifier --------
emotion_classifier = pipeline("text-classification", model="j-hartmann/emotion-english-distilroberta-base", return_all_scores=True)

# -------- 3. Predict emotions --------
def predict_emotions(tweets):
    emotion_results = []
    for tweet in tweets:
        result = emotion_classifier(tweet)[0]
        top_emotion = max(result, key=lambda x: x['score'])
        emotion_results.append((tweet, top_emotion['label'], top_emotion['score']))
    return emotion_results

# -------- 4. Analyze and visualize --------
def analyze_results(emotion_results):
    df = pd.DataFrame(emotion_results, columns=['Tweet', 'Emotion', 'Confidence'])
    emotion_counts = df['Emotion'].value_counts()
    print("Emotion Distribution:\n", emotion_counts)

    # Plot
    emotion_counts.plot(kind='bar', title="Emotion Distribution in Tweets")
    plt.xlabel("Emotion")
    plt.ylabel("Tweet Count")
    plt.show()
    return df

# -------- 5. Run the pipeline --------
if __name__ == "__main__":
    query = "mental health since:2024-12-01 until:2025-01-01"
    tweets = scrape_tweets(query, max_tweets=200)
    emotion_results = predict_emotions(tweets)
    df = analyze_results(emotion_results)
    df.to_csv("emotion_analysis_results.csv", index=False)
    print("Saved to emotion_analysis_results.csv")

Notes

  • Model Used: j-hartmann/emotion-english-distilroberta-base (7 emotion classes: anger, disgust, fear, joy, neutral, sadness, surprise)

  • Data Source: Twitter via snscrape, no need for API keys.

  • You can replace the scraping with your dataset if working with Facebook, Reddit, or Instagram (or for privacy reasons).
